def +(other)
        if (other.instance_of? Fixnum)

        end

        if(@dem == other.dem)

          Fraccion.new(@num * other.dem + @dem * other.num, @dem * other.dem)
        end

def -(other)
        if (other.instance_of? Fixnum)
                return Fraccion.new(@num+(@dem * other),@dem)
        end
        if(@dem == other.dem)
          Fraccion.new(@num - other.num, @dem)
        else
          Fraccion.new(@num * other.dem - @dem * other.num, @dem * other.dem)
        end
end
